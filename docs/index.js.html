import Alleria from './Alleria'
import MinterSetPrice from './MinterSetPrice'
import moment from './moment'
import { ethers } from 'ethers';
import GasPrice from './GasPrice';

/**
 * sdk 的入口文件
 * @module index
 */
export default class index{
    /**
    * 初始化sdk
    * @constructor
    * @param {object} ethereumOrJsonRpc - window.ethereum  或者 jsonrpc 的url 用于walletconnect
    * @param {string} account - 当前账户的地址
    * @param {string} chainID - 当前的网络id
    */
    constructor(ethereumOrJsonRpc, account, chainID) {
        if(typeof ethereumOrJsonRpc == "string"){
            this.library = new ethers.providers.JsonRpcProvider(ethereumOrJsonRpc);
        }else{
            this.library = new ethers.providers.Web3Provider(ethereum||window.ethereum);
        }
        

        this.chainID = chainID;
        this.account= account;
        /**Alleria 实例*/
        this.Alleria = new Alleria(this.library,this.account,this.chainID,false)
        /**MinterSetPrice 实例*/
        this.MinterSetPrice = new MinterSetPrice(this.library,this.account,this.chainID,false)
        /**moment 实例*/
        this.moment= new moment(this.library)
        /**GasPrice 实例*/
        this.GasPrice = new GasPrice(this.library,this.account,this.chainID)
        /**AlleriaGas 实例 Alleria合约相关操作方法获取gas */
        this.AlleriaGas = new Alleria(this.library,this.account,this.chainID,true)
        /**MinterSetPriceGas 实例 MinterSetPrice合约相关操作方法获取gas */
        this.MinterSetPriceGas = new MinterSetPrice(this.library,this.account,this.chainID,true)
        
    }
}
